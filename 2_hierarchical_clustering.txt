rm(list=ls()) 
install.packages ("clValid") 
library (clValid) 
data <- read.csv("USArrests.csv", row.names = 1) 
df <- scale (data) 
dissim <- dist(df, method = 'euclidean') 
hierClust <- hclust(dissim, method = 'complete') 
plot (hierClust) 
cluster <- cutree (hierClust, k = 4) 
dunn (dissim, cluster) 
rect.hclust (hierClust, k = 4, border = 2:4) 
abline (h = 4, col = 'red')
